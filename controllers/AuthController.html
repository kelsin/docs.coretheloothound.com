app/controllers/AuthController.js
'use strict';

Controller for the login/logout view

window.App.AuthController = Ember.Controller.extend({ needs: ['application'], apikey: Ember.computed.alias('controllers.application.apikey'), user: Ember.computed.alias('controllers.application.user'), loggedIn: Ember.computed.alias('controllers.application.loggedIn'), actions: { login: function() { $.ajax({ type: 'POST', url: 'https://api.byfirebepurged.com/login', data:JSON.stringify({ redirect: window.location.protocol + '//' + window.location.host + '/#/apikey/' }), contentType:'application/json', success: function(data) { window.location = data.href; } }); }, logout: function() { this.get('storage').removeValue('apikey'); this.set('apikey', undefined); this.set('user', undefined); this.transitionToRoute('index'); } } });

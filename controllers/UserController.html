window.App.UserController = Ember.ObjectController.extend({ user: Ember.computed.alias('session.user'), init: function() { this.set('realm', 'All'); }, onlyMax: true, updateRealm: function() { var realm = this.get('realm'); if (realm) { window.localStorage.setItem('coretheloothound_realm', realm); } }.observes('realm'), realms: function() { this.set('realm', window.localStorage.getItem('coretheloothound_realm') || 'All'); return ['All'].concat(_.uniq(_.map(this.get('session.user.characters'), function(character) { return character.realm; })).sort()); }.property('session.user.characters.@each.realm'), characters: function() { var realm = this.get('realm'); var onlyMax = this.get('onlyMax'); return _.filter(this.get('user.characters'), function(character) { if(onlyMax && character.level < 90) { return false; } if(!realm || realm === 'All') { return true; } else { return realm === character.realm; } }).sort(function(a, b) { var d = b.level - a.level; if (d === 0) { if(a.name < b.name) return -1; if(a.name > b.name) return 1; return 0; } else { return d; } }); }.property('user.characters.@each', 'realm', 'onlyMax') });

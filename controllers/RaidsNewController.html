window.App.RaidsNewController = Ember.ObjectController.extend({ editDate: function() { return moment(this.get('date')).format('D MMMM, YYYY'); }.property('date'), editTime: function() { return moment(this.get('date')).format('h:mm A'); }.property('date'), updateDate: function() { this.set('date', moment(new Date(this.get('editDate') + ' ' + this.get('editTime')))); }.observes('editDate', 'editTime'), actions: { create: function() { var self = this; var raid = this.get('model'); return raid.save() .then(function(raid) { self.transitionToRoute('raid.index', raid); }) .catch(function(err) { console.log(err); }); } } });

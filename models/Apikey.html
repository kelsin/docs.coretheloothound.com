app/models/Apikey.js
'use strict';

Mixin to allow other objects to store and load the apikey

window.App.Apikey = Ember.Mixin.create({ apikeyItem: 'coretheloothound_apikey', getApikey: function() { var apikeyItem = this.get('apikeyItem'); return window.localStorage.getItem(apikeyItem); }, setApikey: function(apikey) { var apikeyItem = this.get('apikeyItem'); return window.localStorage.setItem(apikeyItem, apikey); }, removeApikey: function() { var apikeyItem = this.get('apikeyItem'); return window.localStorage.removeItem(apikeyItem); } });

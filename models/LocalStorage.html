app/models/LocalStorage.js
'use strict'; window.App.LocalStorage = Ember.Object.extend({ get: function(key) { if (Modernizr.localstorage) { return window.localStorage.getItem(key); } else { return Ember.$.cookie(key); } }, set: function(key, value) { if (Modernizr.localstorage) { window.localStorage.setItem(key, value); } else { Ember.$.cookie(key, value); } }, remove: function(key) { if (Modernizr.localstorage) { window.localStorage.removeItem(key); } else { Ember.$.removeCookie(key); } } });

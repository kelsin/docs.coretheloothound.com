When the apikey updates, lets fetch a new user object

updateUser: function() { var self = this; var apikey = this.get('apikey'); if (apikey) { Ember.$.ajax({ type: 'GET', url: 'https://api.byfirebepurged.com/user', headers: { Authorization: 'apikey ' + apikey }, contentType:'application/json', success: function(data) { self.set('user', data.user); } }); } else { self.set('user', undefined); } }.observes('apikey') });

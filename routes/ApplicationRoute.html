app/routes/ApplicationRoute.js
'use strict';

Handles loading the user for the application

window.App.ApplicationRoute = Ember.Route.extend({ model: function() { var apikey = this.get('storage').getValue('apikey'); if (apikey) { return new Ember.RSVP.Promise(function(resolve, reject) { Ember.$.ajax({ type: 'GET', url: 'https://api.byfirebepurged.com/user', headers: { Authorization: 'apikey ' + apikey }, contentType:'application/json', success: function(data) { resolve({ apikey: apikey, user: data.user }); }, error: reject }); }); } else { return Ember.RSVP.resolve({ apikey: undefined, user: undefined }); } }, setupController: function(controller, model) { controller.set('apikey', model.apikey); controller.set('user', model.user); }, actions: { loadUser: function() { console.log('Refreshing'); this.refresh(); } } });
